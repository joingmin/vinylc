// vcui 프레임워크에서 breakpoint를 지날 때마다 breakpointchange 라는 이벤트를 발생시켜 준다.
// 스크립트단에서 모바일/PC에 따라 처리해야 하는 작업이 있을 경우 이 이벤트를 이용하면 된다.

$(window).on('breakpointchange', function (e) {
    console.log(window.breakpoint); // window.breakpoint객체에 현재 해상도에 대한 정보가 담겨있다.
    $('body').css('backgroundColor', window.breakpoint.isMobile ? 'red' : 'blue');
});
